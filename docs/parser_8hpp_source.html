35  parser(int argument_count, char** argument_values, const char* options) {
36  for (int i = 0; i < argument_count; ++i) {
37  const int value_length = strlen(argument_values[i]);
38  for (int j = 0; j < value_length; ++j) {
39  const char current_char = argument_values[i][j];
40  const char next_char = argument_values[i][j + 1];
41 
42  if (current_char == '-' &&
43  j <= value_length &&
44  (!strcmp(options, "*") || !strchr(options, next_char))) {
45  // ^ might cause bug, changed strchr() != NULL to !strchr(); check git
46  option opt = {"", ""};
47 
48  // Parse option name
49  for (int k = 1; k < value_length; ++k) {
50  const char current_char_k = argument_values[i][k];
51  if ((current_char_k >= 65 && current_char_k <= 90) || (current_char_k >= 97 && current_char_k <= 122)) {
52  opt.name += current_char_k;
53  } else { // space; indicates beginning of value
54  break;
55  }
56  }
57 
58  // Parse option value
59  for (int k = 0; k < strlen(argument_values[i + 1]); ++k) {
60  const char current_char_k = argument_values[i + 1][k];
61  if (current_char_k != 32) {
62  opt.value += current_char_k;
63  } else { // space; indicates end of value
64  break;
65  }
66  }
67  // Check if option name is empty
68  // This may be because argument_values has - in it followed by a character that is not A-Za-z
69  if (opt.name == "") throw std::runtime_error("Invalid option name");
70 
71  this->options.push_back(opt);
72  }
73  }
74  }
